  btn.onClick.listen((e) => msg.text = 'Dart!');

Built-in libraries


dart:io

import 'dart:io';

main() {
  HttpServer.bind("127.0.0.1", 8888).then((svr) {
    svr.listen((req) {
      req.response.write("Hello World\n");
      req.response.close();
    });

    print("Listening at http://127.0.0.1:8888/");
  });
}
